```typescript
/**
 * MinerU 云端 API 解析器
 * 调用 mineru.net 精准解析 API (v4) 解析 PDF 文档
 * Per D-02: priority=2, canParse 检测扫描版 PDF
 */

import { readFile } from 'node:fs/promises';
import { createHash } from 'node:crypto';
import type { DocumentParser, ParseInput, ParseOptions, ParseResult, ParseWarning } from '../types.js';
import type { DocumentAst } from '../../document-ast.js';
import { mapContentListToAst, type ContentListItem } from './mapper.js';

const MINERU_API_URL = 'https://mineru.net/api/v4/extract/task';
const MINERU_POLL_INTERVAL_MS = 2000;

/**
 * 检测 PDF 是否为扫描版（文本层极少或为空）
 * 读取前几页的文本内容，如果字符数很少则判定为扫描版
 */
async function isScannedPdf(filePath: string): Promise<boolean> {
  try {
    const { PDFParse } = await import('pdf-parse');
    const buffer = await readFile(filePath);
    const parser = new PDFParse({ data: new Uint8Array(buffer) });
    const result = await parser.getText({ lineEnforce: false });
    await parser.destroy();

    // 检查前 3 页的文本量
    const pagesToCheck = result.pages.slice(0, 3);
    let totalChars = 0;
    for (const page of pagesToCheck) {
      totalChars += (page.text || '').trim().length;
    }

    // 前 3 页平均少于 50 字符/页 → 扫描版
    const avgCharsPerPage = totalChars / Math.max(pagesToCheck.length, 1);
    return avgCharsPerPage < 50;
  } catch {
    // pdf-parse 失败 → 无法判断，返回 false 让调用方决定
    return false;
  }
}

export class MinerUParser implements DocumentParser {
  readonly id = 'mineru-parser';
  readonly name = 'MinerU 云端解析器';
  readonly supportedExtensions = ['.pdf'];
  readonly mimeTypes = ['application/pdf'];
  readonly priority = 2; // Per D-02: 低于 pdf-parse (priority=1)

  private apiToken: string;

  constructor(apiToken: string) {
    this.apiToken = apiToken;
  }

  async canParse(input: ParseInput): Promise<boolean> {
    if (!input.fileName.toLowerCase().endsWith('.pdf')) return false;
    // Per D-02: 检测扫描版 PDF 特征
    return isScannedPdf(input.filePath);
  }

  async parse(input: ParseInput, options: ParseOptions): Promise<ParseResult> {
    const startTime = Date.now();
    const warnings: ParseWarning[] = [];

    try {
      // 1. 读取文件
      const fileBuffer = await readFile(input.filePath);
      const sha256 = createHash('sha256').update(fileBuffer).digest('hex');

      // 2. 调用 MinerU API 创建任务
      const taskId = await this.createTask(fileBuffer, input.fileName);
      warnings.push({ code: 'MINERU_TASK', message: `MinerU task created: ${taskId}`, severity: 'info' });

      // 3. 轮询等待完成
      const result = await this.pollTask(taskId, options.timeout);

      // 4. 解析 content_list.json
      const contentList = result as ContentListItem[];
      const blocks = mapContentListToAst(contentList);

      // 5. 统计
      let wordCount = 0;
      const countWords = (text: string) => {
        const cn = (text.match(/[一-鿿]/g) || []).length;
        const en = text.replace(/[一-鿿]/g, '').split(/\s+/).filter(w => w.length > 0).length;
        return cn + en;
      };
      for (const block of blocks) {
        if (block.type === 'paragraph') wordCount += countWords(block.text);
        else if (block.type === 'table') wordCount += block.rows.flat().reduce((s, c) => s + countWords(c), 0);
        else if (block.type === 'section') {
          wordCount += countWords(block.title);
          // count children recursively (simplified)
        }
      }

      // 6. 构建 DocumentAst
      const ast: DocumentAst = {
        id: sha256,
        filename: input.fileName,
        sha256,
        pageCount: null, // MinerU 不直接返回总页数
        wordCount,
        parserVersion: 'mineru-api-v4',
        blocks,
      };

      return {
        success: true,
        ast,
        warnings,
        duration: Date.now() - startTime,
        parserId: this.id,
      };
    } catch (error) {
      return {
        success: false,
        warnings,
        duration: Date.now() - startTime,
        parserId: this.id,
        error: {
          code: 'MINERU_ERROR',
          message: error instanceof Error ? error.message : 'Unknown MinerU error',
        },
      };
    }
  }

  private async createTask(fileBuffer: Buffer, fileName: string): Promise<string> {
    // 创建 multipart/form-data 请求
    const boundary = `----FormBoundary${Date.now()}`;
    const filePart = [
      `--${boundary}`,
      `Content-Disposition: form-data; name="file"; filename="${fileName}"`,
      `Content-Type: application/pdf`,
      '',
      fileBuffer,
    ].join('\r\n');

    const body = Buffer.concat([
      Buffer.from(filePart + '\r\n'),
      Buffer.from(`--${boundary}--\r\n`),
    ]);

    const response = await fetch(MINERU_API_URL, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${this.apiToken}`,
        'Content-Type': `multipart/form-data; boundary=${boundary}`,
      },
      body,
    });

    if (!response.ok) {
      const text = await response.text();
      throw new Error(`MinerU API error ${response.status}: ${text}`);
    }

    const data = await response.json() as { code: number; msg: string; data: { task_id: string } };
    if (data.code !== 0) {
      throw new Error(`MinerU task creation failed: ${data.msg}`);
    }

    return data.data.task_id;
  }

  private async pollTask(taskId: string, timeoutMs: number): Promise<unknown> {
    const deadline = timeoutMs > 0 ? Date.now() + timeoutMs : Infinity;

    while (Date.now() < deadline) {
      const response = await fetch(`${MINERU_API_URL}/${taskId}`, {
        headers: { 'Authorization': `Bearer ${this.apiToken}` },
      });

      if (!response.ok) {
        throw new Error(`MinerU poll error ${response.status}`);
      }

      const data = await response.json() as {
        code: number;
        data: { state: string; content_list?: unknown };
      };

      if (data.data.state === 'complete') {
        return data.data.content_list;
      }

      if (data.data.state === 'failed') {
        throw new Error('MinerU task failed');
      }

      // 等待后重试
      await new Promise(r => setTimeout(r, MINERU_POLL_INTERVAL_MS));
    }

    throw new Error('MinerU task timeout');
  }
}
```
