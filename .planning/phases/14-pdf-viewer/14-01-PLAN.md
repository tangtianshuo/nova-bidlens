   - Add a public method to the `RiskReviewService` class:
     ```typescript
     getPdfFile(projectId: string, submissionId: string): { filePath: string } | null {
       const submissionRows = this.submissionRepo.getByProject(projectId);
       const sub = submissionRows.find(s => s.id === submissionId);
       if (!sub || !sub.file_path_encrypted) return null;
       const filePath = decrypt(sub.file_path_encrypted, this.encryptionKey);
       // Per D-04: path validation to prevent path traversal
       const resolvedPath = path.resolve(filePath);
       if (!fs.existsSync(resolvedPath)) return null;
       return { filePath: resolvedPath };
     }
     ```
     Note: `path` is already imported at the top of the file (`import path from 'node:path'`). Add `import fs from 'node:fs'` at the top if not already present. `decrypt` is already imported from `../db/crypto.js`.
