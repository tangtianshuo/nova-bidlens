## 12.1 构建流水线

### 12.1.1 Monorepo 构建架构

BidLens 采用 pnpm workspace 管理的 Monorepo 结构，包含三个主要子项目：

```
bidlens/
├── package.json              # 根 package.json（workspace 配置）
├── pnpm-workspace.yaml       # workspace 定义
├── apps/
│   └── desktop/              # Electron 应用（前端 + 主进程）
│       ├── package.json
│       ├── electron/         # 主进程代码
│       ├── src/              # React 渲染进程代码
│       ├── dist/             # 前端构建产物
│       └── release/          # Electron 打包产物
├── packages/
│   ├── parser-docx/          # Word 解析器包
│   ├── parser-pdf/           # PDF 解析器包
│   ├── core-bridge/          # Node ↔ Rust 通信层
│   └── shared-types/         # 共享类型定义
├── engine/                   # Rust 核心引擎
│   ├── Cargo.toml
│   ├── core/                 # 纯 Rust crate
│   └── bridge/               # JSON-RPC 通信层
└── scripts/                  # 构建脚本
    ├── build.sh              # 全量构建
    ├── build-engine.sh       # Rust 引擎构建
    └── package.sh            # 打包脚本
```

**pnpm-workspace.yaml：**

```yaml
packages:
  - 'apps/*'
  - 'packages/*'
```

### 12.2.3 安装包体积优化

```
优化前                                    优化后
─────────────────────────                 ─────────────────────────
node_modules:  380MB                     精简依赖:     120MB
Rust binary:   85MB                      UPX 压缩:     35MB
ONNX models:   130MB                     按需下载:     0MB (首次启动下载)
静态资源:       15MB                      Tree-shake:   8MB
─────────────────────────                 ─────────────────────────
总计:          610MB                     总计:         163MB

NSIS 安装包:   ~180MB                    NSIS 安装包:  ~65MB
```

**体积优化措施：**

1. **依赖精简**：移除未使用的依赖，用轻量库替代重型库
2. **Rust 二进制压缩**：使用 UPX 压缩 Rust 二进制（85MB → 35MB）
3. **模型按需下载**：ONNX 模型不在安装包中，首次使用时下载
4. **Tree-shaking**：Vite 自动移除未使用的代码
5. **资源压缩**：图片使用 WebP 格式，字体子集化

### 12.3.2 electron-updater 集成

```typescript
// electron/main/updater.ts
import { autoUpdater } from 'electron-updater';
import log from 'electron-log';

class AppUpdater {
  constructor() {
    // 配置日志
    autoUpdater.logger = log;

    // 更新服务器配置
    autoUpdater.autoDownload = false;   // 手动控制下载
    autoUpdater.autoInstallOnAppQuit = true;
    autoUpdater.allowDowngrade = false;  // 禁止降级

    // 更新事件监听
    autoUpdater.on('checking-for-update', () => {
      log.info('Checking for update...');
    });

    autoUpdater.on('update-available', (info) => {
      log.info('Update available:', info.version);
      // 通知渲染进程
      this.notifyRenderer('update:available', {
        version: info.version,
        releaseDate: info.releaseDate,
        releaseNotes: info.releaseNotes,
      });
    });

    autoUpdater.on('download-progress', (progress) => {
      this.notifyRenderer('update:progress', {
        percent: progress.percent,
        bytesPerSecond: progress.bytesPerSecond,
        transferred: progress.transferred,
        total: progress.total,
      });
    });

    autoUpdater.on('update-downloaded', (info) => {
      log.info('Update downloaded:', info.version);
      this.notifyRenderer('update:ready', { version: info.version });
    });

    autoUpdater.on('error', (error) => {
      log.error('Update error:', error);
      this.notifyRenderer('update:error', { message: error.message });
    });
  }

  /** 检查更新 */
  async checkForUpdates(): Promise<void> {
    try {
      await autoUpdater.checkForUpdates();
    } catch (error) {
      log.error('Failed to check for updates:', error);
    }
  }

  /** 下载更新 */
  async downloadUpdate(): Promise<void> {
    await autoUpdater.downloadUpdate();
  }

  /** 安装更新并重启 */
  quitAndInstall(): void {
    autoUpdater.quitAndInstall(false, true);
  }

  /** 通知渲染进程 */
  private notifyRenderer(channel: string, data: unknown): void {
    BrowserWindow.getAllWindows().forEach(win => {
      win.webContents.send(channel, data);
    });
  }
}
```

### 12.3.3 更新检查策略

```typescript
/**
 * 更新检查时机：
 *
 * 1. 应用启动时 — 延迟 30 秒后检查（避免影响启动速度）
 * 2. 定时检查 — 每 4 小时检查一次
 * 3. 手动检查 — 用户点击"检查更新"按钮
 */
class UpdateScheduler {
  private checkInterval: NodeJS.Timeout | null = null;

  start(): void {
    // 启动后延迟 30 秒检查
    setTimeout(() => {
      appUpdater.checkForUpdates();
    }, 30 * 1000);

    // 每 4 小时定时检查
    this.checkInterval = setInterval(() => {
      appUpdater.checkForUpdates();
    }, 4 * 60 * 60 * 1000);
  }

  stop(): void {
    if (this.checkInterval) {
      clearInterval(this.checkInterval);
    }
  }
}
```

### 12.3.4 更新回滚机制

```typescript
/**
 * 更新回滚方案
 *
 * 策略：保留最近 2 个版本的安装包，更新失败时可回退
 *
 * 流程：
 * 1. 下载新版本前，备份当前版本信息
 * 2. 安装新版本后，记录安装状态为 "pending"
 * 3. 用户首次启动新版本，确认正常后标记为 "confirmed"
 * 4. 若新版本启动失败（崩溃检测），自动回滚到上一版本
 */
class UpdateRollback {
  /** 备份当前版本信息 */
  async backupCurrentVersion(): Promise<void> {
    const currentVersion = app.getVersion();
    await settings.set('update:previous_version', currentVersion);
    await settings.set('update:rollback_state', 'pending');
  }

  /** 确认新版本正常 */
  async confirmUpdate(): Promise<void> {
    await settings.set('update:rollback_state', 'confirmed');
  }

  /** 检查是否需要回滚 */
  async checkRollbackNeeded(): Promise<boolean> {
    const state = await settings.get('update:rollback_state');
    if (state === 'pending') {
      // 新版本尚未确认，且启动了崩溃检测
      const crashCount = await settings.get('update:crash_count', 0);
      if (crashCount >= 2) {
        return true; // 连续崩溃 2 次，需要回滚
      }
    }
    return false;
  }

  /** 执行回滚 */
  async rollback(): Promise<void> {
    const previousVersion = await settings.get('update:previous_version');

    // ...（通过更新服务器获取指定版本的安装包）
  }
}
```

## 12.4 CI/CD 流水线

### 12.4.1 GitHub Actions 工作流

```yaml
# .github/workflows/ci.yml
name: CI/CD Pipeline

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]
  tags:
    - 'v*'

env:
  NODE_VERSION: '20'
  RUST_VERSION: 'stable'
  PNPM_VERSION: '9'

jobs:
  # ─── PR 检查 ───────────────────────────────────────────
  check:
    name: Lint & Type Check
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: pnpm/action-setup@v4
        with:
          version: ${{ env.PNPM_VERSION }}

      - uses: actions/setup-node@v4
        with:
          node-version: ${{ env.NODE_VERSION }}
          cache: 'pnpm'

      - run: pnpm install --frozen-lockfile

      # TypeScript 类型检查
      - name: Type Check
        run: pnpm typecheck

      # ESLint 代码检查
      - name: Lint
        run: pnpm lint

      # Rust 代码检查
      - name: Rust Clippy
        uses: actions-rs/clippy-check@v1
        with:
          toolchain: ${{ env.RUST_VERSION }}
          args: --manifest-path engine/Cargo.toml --all-targets -- -D warnings

  # ─── 测试 ─────────────────────────────────────────────
  test:
    name: Test
    needs: check
    strategy:
      matrix:
        os: [ubuntu-latest, windows-latest, macos-latest]
    runs-on: ${{ matrix.os }}
    steps:
      - uses: actions/checkout@v4

      - uses: pnpm/action-setup@v4
        with:
          version: ${{ env.PNPM_VERSION }}

      - uses: actions/setup-node@v4
        with:
          node-version: ${{ env.NODE_VERSION }}
          cache: 'pnpm'

      - uses: dtolnay/rust-toolchain@stable
        with:
          toolchain: ${{ env.RUST_VERSION }}

      - run: pnpm install --frozen-lockfile

      # 单元测试
      - name: Unit Tests
        run: pnpm test

      # Rust 测试
      - name: Rust Tests
        run: cd engine && cargo test --release

      # E2E 测试（仅 Linux）
      - name: E2E Tests
        if: matrix.os == 'ubuntu-latest'
        run: pnpm test:e2e

  # ─── 主分支构建 ────────────────────────────────────────
  build:
    name: Build
    needs: test
    if: github.ref == 'refs/heads/main' || startsWith(github.ref, 'refs/tags/')
    strategy:
      matrix:
        include:
          - os: ubuntu-latest
            platform: linux
            target: [x64]
          - os: windows-latest
            platform: win
            target: [x64]
          - os: macos-latest
            platform: mac
            target: [x64, arm64]
    runs-on: ${{ matrix.os }}
    steps:
      - uses: actions/checkout@v4

      - uses: pnpm/action-setup@v4
        with:
          version: ${{ env.PNPM_VERSION }}

      - uses: actions/setup-node@v4
        with:
          node-version: ${{ env.NODE_VERSION }}
          cache: 'pnpm'

      - uses: dtolnay/rust-toolchain@stable

      - run: pnpm install --frozen-lockfile

      # 构建前端
      - name: Build Frontend
        run: pnpm build:desktop

      # 构建 Rust 引擎
      - name: Build Engine
        run: cd engine && cargo build --release

      # Electron 打包
      - name: Package Electron
        run: pnpm package:${{ matrix.platform }}
        env:
          # Windows 签名
          WIN_CSC_LINK: ${{ secrets.WIN_CSC_LINK }}
          WIN_CSC_KEY_PASSWORD: ${{ secrets.WIN_CSC_KEY_PASSWORD }}
          # macOS 签名
          CSC_LINK: ${{ secrets.MAC_CSC_LINK }}
          CSC_KEY_PASSWORD: ${{ secrets.MAC_CSC_KEY_PASSWORD }}
          APPLE_ID: ${{ secrets.APPLE_ID }}
          APPLE_ID_PASSWORD: ${{ secrets.APPLE_ID_PASSWORD }}
          APPLE_TEAM_ID: ${{ secrets.APPLE_TEAM_ID }}

      # 上传构建产物
      - name: Upload Artifacts
        uses: actions/upload-artifact@v4
        with:
          name: release-${{ matrix.platform }}
          path: apps/desktop/release/**/*
          retention-days: 30

  # ─── Release 发布 ──────────────────────────────────────
  release:
    name: Release
    needs: build
    if: startsWith(github.ref, 'refs/tags/')
    runs-on: ubuntu-latest
    permissions:
      contents: write
    steps:
      - uses: actions/checkout@v4

      # 下载所有平台的构建产物
      - name: Download Artifacts
        uses: actions/download-artifact@v4
        with:
          path: artifacts

      # 创建 GitHub Release
      - name: Create Release
        uses: softprops/action-gh-release@v2
        with:
          files: artifacts/**/*
          draft: false
          prerelease: false
          generate_release_notes: true

      # 上传到更新服务器
      - name: Upload to Update Server
        run: |
          # 将安装包和 latest.yml/latest-mac.yml/latest-linux.yml
          # 上传到 CDN 更新服务器
          ./scripts/upload-release.sh
```

### 12.6.2 Rust 子进程崩溃恢复

```typescript
/**
 * Rust 子进程健康监控与自动恢复
 *
 * 策略：
 * 1. 每 30 秒发送心跳
 * 2. 连续 3 次无响应则判定为崩溃
 * 3. 自动重启子进程
 * 4. 重启失败超过 5 次则提示用户
 */
class RustProcessManager {
  private crashCount = 0;
  private maxRestarts = 5;
  private heartbeatInterval: NodeJS.Timeout | null = null;

  async spawn(): Promise<void> {
    this.process = spawn(this.binaryPath, [], {
      stdio: ['pipe', 'pipe', 'pipe'],
      env: { ...process.env, RUST_LOG: 'info' },
    });

    this.process.on('exit', (code, signal) => {
      log.error(`Rust process exited: code=${code}, signal=${signal}`);
      this.onProcessExit(code);
    });

    this.process.stderr?.on('data', (data) => {
      log.error(`[Engine STDERR] ${data.toString()}`);
    });

    // 启动心跳
    this.startHeartbeat();
  }

  private onProcessExit(code: number | null): void {
    this.stopHeartbeat();
    this.crashCount++;

    if (this.crashCount <= this.maxRestarts) {
      log.warn(`Restarting Rust process (attempt ${this.crashCount}/${this.maxRestarts})`);
      setTimeout(() => this.spawn(), 1000);
    } else {
      log.error('Rust process crashed too many times');
      // 通知用户
      dialog.showErrorBox(
        '引擎错误',
        '文档处理引擎多次崩溃，请尝试重启应用或联系技术支持。'
      );
    }
  }

  private startHeartbeat(): void {
    this.heartbeatInterval = setInterval(async () => {
      try {
        await this.rpc.call('ping', [], 5000); // 5 秒超时
        this.consecutiveFailures = 0;
      } catch {
        this.consecutiveFailures++;
        if (this.consecutiveFailures >= 3) {
          log.error('Rust process heartbeat failed 3 times, killing...');
          this.process?.kill();
        }
      }
    }, 30000);
  }
}
```

### 12.6.3 数据库损坏恢复

```typescript
/**
 * SQLite 数据库完整性检查与恢复
 */
class DatabaseRecovery {
  async checkIntegrity(dbPath: string): Promise<boolean> {
    const db = await open({ filename: dbPath, driver: sqlite3.Database });
    try {
      const result = await db.get('PRAGMA integrity_check');
      return result.integrity_check === 'ok';
    } finally {
      await db.close();
    }
  }

  async recover(dbPath: string): Promise<boolean> {
    log.warn('Attempting database recovery...');

    // 1. 备份损坏的数据库
    const backupPath = `${dbPath}.corrupted.${Date.now()}`;
    await fs.promises.copyFile(dbPath, backupPath);

    // 2. 尝试导出数据
    try {
      const db = await open({ filename: dbPath, driver: sqlite3.Database, mode: sqlite3.OPEN_READONLY });
      const tables = await db.all("SELECT name FROM sqlite_master WHERE type='table'");

      for (const table of tables) {
        try {
          await db.all(`SELECT * FROM ${table.name} LIMIT 1`);
          log.info(`Table ${table.name} is readable`);
        } catch {
          log.warn(`Table ${table.name} is corrupted`);
        }
      }
      await db.close();
    } catch {
      log.error('Cannot open corrupted database at all');
    }

    // 3. 删除损坏的数据库，创建新的
    await fs.promises.unlink(dbPath);
    await this.initializeNewDatabase(dbPath);

    log.info('Database recovery completed (data may be lost)');
    return true;
  }
}
```

```typescript
      up: `
        CREATE TABLE IF NOT EXISTS plugins (
          id TEXT PRIMARY KEY,
          name TEXT NOT NULL,
          version TEXT NOT NULL,
          type TEXT NOT NULL,
          status TEXT DEFAULT 'inactive',
          installed_at INTEGER,
          manifest_json TEXT
        );
      `,
```

### 12.7.3 AST Schema 版本兼容

```typescript
/**
 * AST Schema 版本兼容策略
 *
 * 规则：
 * - MAJOR 版本变更：不兼容，旧 AST 需要重新解析
 * - MINOR 版本变更：向后兼容，旧 AST 可读取，新 AST 可能包含新字段
 * - PATCH 版本变更：完全兼容
 */
interface ASTVersionInfo {
  version: string;       // e.g., "1.2.0"
  min_compatible: string; // e.g., "1.0.0" — 可读取的最低版本
}

function isASTCompatible(astVersion: string, appVersion: string): boolean {
  const astSemver = semver.parse(astVersion);
  const appSemver = semver.parse(appVersion);

  if (!astSemver || !appSemver) return false;

  // MAJOR 版本必须匹配
  return astSemver.major === appSemver.major;
}
```
