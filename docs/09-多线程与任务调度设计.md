## 9.2 Tokio 异步运行时设计

### 9.2.1 运行时配置

```rust
use tokio::runtime::Builder;

fn create_runtime() -> tokio::runtime::Runtime {
    Builder::new_multi_thread()
        .worker_threads(4)           // IO worker 线程数（默认 = CPU 核心数，这里固定为 4）
        .max_blocking_threads(64)    // 最大阻塞线程数（用于 spawn_blocking）
        .thread_name("bidlens-io")   // 线程名前缀，方便调试
        .thread_stack_size(4 * 1024 * 1024)  // 每线程 4MB 栈空间
        .enable_all()                // 启用 IO 和 time 驱动
        .build()
        .expect("Failed to create Tokio runtime")
}
```

**线程数配置策略**：

```
设备类型          IO Workers    Blocking Threads    Rayon Threads
─────────────────────────────────────────────────────────────
低配 (4核)         2             32                  2
中配 (8核)         4             64                  4
高配 (16核+)       4             64                  8

设计原则:
- IO Workers: 固定 4 个即可（IO 操作不密集）
- Blocking Threads: 上限 64（ONNX 推理可能并发多个）
- Rayon: CPU 核心数 / 2（留一半给系统和 Electron）
```

### 9.2.2 异步任务类型划分

```rust
// IO 密集型任务: 使用 tokio::spawn
// 特点: 大部分时间在等待 IO，CPU 占用低
async fn handle_request(request: JsonRpcRequest) -> JsonRpcResponse {
    // 读取 stdin、写入 stdout、HTTP 请求等
    tokio::spawn(async move {
        match request.method.as_str() {
            "ping" => handle_ping(request).await,
            "health" => handle_health(request).await,
            "cancel" => handle_cancel(request).await,
            _ => dispatch_to_engine(request).await,
        }
    }).await.unwrap()
}

// CPU 密集型任务: 使用 spawn_blocking 或 Rayon
// 特点: 持续占用 CPU，不能阻塞 IO worker
async fn heavy_compute(data: Vec<f32>) -> Vec<f32> {
    tokio::task::spawn_blocking(move || {
        // 在阻塞线程池中执行 CPU 密集计算
        // 不会阻塞 Tokio IO worker
        compute_embedding(&data)
    }).await.unwrap()
}
```

### 9.2.3 任务 spawn 策略

```
决策树: 如何选择正确的 spawn 方式

任务类型判断
    │
    ├── 主要是 IO 操作（文件读写、网络请求、等待子进程）
    │   → tokio::spawn（在 IO worker 上执行）
    │
    ├── 主要是 CPU 计算（向量运算、矩阵乘法、文本处理）
    │   │
    │   ├── 计算量小（< 1ms）
    │   │   → 直接在当前 task 执行
    │   │
    │   ├── 计算量中等（1ms - 100ms）
    │   │   → tokio::task::spawn_blocking（阻塞线程池）
    │   │
    │   └── 计算量大（> 100ms）且可并行
    │       → rayon::spawn（Rayon 线程池）
    │
    └── 混合型（先 IO 再 CPU）
        → tokio::spawn 包装，内部 CPU 部分用 spawn_blocking
```

### 9.2.4 CancellationToken 传播

```rust
use tokio_util::sync::CancellationToken;
use std::sync::Arc;

/// 任务管理器，管理所有运行中任务的生命周期
struct TaskManager {
    tasks: DashMap<String, CancellationToken>,
}

impl TaskManager {
    /// 注册新任务
    fn register_task(&self, task_id: String) -> CancellationToken {
        let token = CancellationToken::new();
        self.tasks.insert(task_id, token.clone());
        token
    }

    /// 取消任务
    fn cancel_task(&self, task_id: &str) -> bool {
        if let Some(token) = self.tasks.get(task_id) {
            token.cancel();
            true
        } else {
            false
        }
    }

    /// 任务完成时清理
    fn complete_task(&self, task_id: &str) {
        self.tasks.remove(task_id);
    }
}

/// 带取消传播的 compare 任务
async fn compare_with_cancel(
    task_mgr: Arc<TaskManager>,
    task_id: String,
    params: CompareParams,
) -> Result<CompareResult, AppError> {
    let token = task_mgr.register_task(task_id.clone());

    let result = tokio::select! {
        // 正常执行
        result = compare(params, token.clone()) => result,
        // 取消信号
        _ = token.cancelled() => {
            Err(AppError::Cancelled(task_id))
        }
    };

    task_mgr.complete_task(&task_id);
    result
}
```

### 9.3.2 Embedding 批量推理并行化

```rust
/// 批量 Embedding 推理，利用 Rayon 并行化 batch 处理
fn parallel_embed(
    engine: &EmbeddingEngine,
    chunks: &[Chunk],
    batch_size: usize,
) -> Vec<Vec<f32>> {
    // 将 chunks 分成 batch
    let batches: Vec<&[Chunk]> = chunks.chunks(batch_size).collect();

    // 并行处理每个 batch
    let embeddings: Vec<Vec<Vec<f32>>> = batches.par_iter()
        .map(|batch| {
            let texts: Vec<String> = batch.iter().map(|c| c.text.clone()).collect();
            engine.embed_batch(&texts).unwrap()
        })
        .collect();

    // 展平结果
    embeddings.into_iter().flatten().collect()
}
```

### 9.3.3 向量相似度计算并行化

```rust
/// 并行计算相似度矩阵
fn parallel_similarity_matrix(
    embeddings_a: &[Vec<f32>],
    embeddings_b: &[Vec<f32>],
) -> Vec<Vec<f32>> {
    // 每行（A 的每个 chunk）并行计算
    embeddings_a.par_iter()
        .map(|a_vec| {
            embeddings_b.iter()
                .map(|b_vec| cosine_similarity(a_vec, b_vec))
                .collect()
        })
        .collect()
}

/// 并行 reduce: 从相似度矩阵中提取匹配对
fn parallel_find_matches(
    sim_matrix: &[Vec<f32>],
    threshold: f32,
) -> Vec<(usize, usize, f32)> {
    sim_matrix.par_iter().enumerate()
        .flat_map(|(i, row)| {
            row.iter().enumerate()
                .filter(|(_, &sim)| sim >= threshold)
                .map(|(j, &sim)| (i, j, sim))
                .collect::<Vec<_>>()
        })
        .collect()
}
```

### 9.3.4 Rayon 与 Tokio 的协作

```rust
/// 桥接 Tokio 和 Rayon 的模式
/// Tokio 负责调度，Rayon 负责计算

async fn compare(params: CompareParams) -> Result<CompareResult> {
    // 阶段 1-2: 在 Tokio 中执行 IO 和调度
    let (ast_a, ast_b) = parse_documents(&params).await?;
    let chunks_a = chunk_document(&ast_a, &params.options);
    let chunks_b = chunk_document(&ast_b, &params.options);

    // 阶段 3: 在 Rayon 中执行 CPU 密集的 Embedding
    let embeddings = tokio::task::spawn_blocking(move || {
        let pool = rayon::ThreadPoolBuilder::new()
            .num_threads(num_cpus::get() / 2)
            .build()
            .unwrap();

        pool.install(|| {
            let emb_a = parallel_embed(&engine, &chunks_a, 32);
            let emb_b = parallel_embed(&engine, &chunks_b, 32);
            (emb_a, emb_b)
        })
    }).await?;

    // 阶段 4: 在 Rayon 中执行相似度计算
    let pairs = tokio::task::spawn_blocking(move || {
        parallel_find_matches(&sim_matrix, threshold)
    }).await?;

    // 阶段 5: 回到 Tokio 处理结果和上报
    Ok(CompareResult { pairs, stats })
}
```

### 9.4.2 任务队列设计

```rust
/// 单任务执行器: 同一时间只执行一个比对任务
struct TaskQueue {
    queue: VecDeque<String>,                  // 等待队列
    running: Option<String>,                  // 当前运行的任务 ID
    tasks: DashMap<String, Task>,             // 所有任务的存储
    notify: Arc<Notify>,                      // 新任务通知
}

impl TaskQueue {
    /// 提交新任务
    async fn submit(&self, task: Task) -> TaskHandle {
        let id = task.id.clone();
        self.tasks.insert(id.clone(), task);

        if self.running.is_some() {
            // 已有任务在运行，加入队列
            self.queue.push_back(id.clone());
        } else {
            // 没有任务在运行，直接执行
            self.running = Some(id.clone());
            self.execute_task(&id).await;
        }

        TaskHandle { id, queue: self.clone() }
    }

    /// 执行任务
    async fn execute_task(&self, task_id: &str) {
        let task = self.tasks.get(task_id).unwrap().clone();
        task.state = TaskState::Running;
        task.started_at = Some(Instant::now());

        let result = compare_with_cancel(
            self.task_mgr.clone(),
            task.id.clone(),
            task.params.clone(),
        ).await;

        match result {
            Ok(result) => {
                task.state = TaskState::Completed;
                task.result = Some(serde_json::to_value(result).unwrap());
            }
            Err(AppError::Cancelled(_)) => {
                task.state = TaskState::Cancelled;
            }
            Err(e) => {
                task.state = TaskState::Failed;
                task.error = Some(e);
            }
        }

        task.completed_at = Some(Instant::now());
        self.running = None;

        // 执行队列中的下一个任务
        if let Some(next_id) = self.queue.pop_front() {
            self.execute_task(&next_id).await;
        }
    }
}
```

### 9.4.3 多阶段进度聚合

```rust
/// 进度报告器，聚合多个阶段的进度
struct ProgressReporter {
    task_id: String,
    sender: mpsc::Sender<ProgressEvent>,
}

impl ProgressReporter {
    fn report(&self, phase: &str, current: usize, total: usize) {
        let phase_weight = match phase {
            "parse" => 10,     // 解析占 10%
            "chunk" => 5,      // 分块占 5%
            "embed" => 60,     // Embedding 占 60%
            "match" => 15,     // 匹配占 15%
            "diff" => 10,      // Diff 占 10%
            _ => 0,
        };

        let phase_progress = (current as f64 / total as f64) * phase_weight as f64;
        let cumulative = self.get_cumulative_progress(phase) + phase_progress as usize;

        let event = ProgressEvent {
            task_id: self.task_id.clone(),
            phase: phase.to_string(),
            current,
            total,
            percent: cumulative.min(100),
            message: format!("{} ({}/{})", phase, current, total),
        };

        // 非阻塞发送（如果接收端慢，丢弃旧事件）
        let _ = self.sender.try_send(event);
    }

    fn get_cumulative_progress(&self, current_phase: &str) -> usize {
        let phases = ["parse", "chunk", "embed", "match", "diff"];
        let weights = [10, 5, 60, 15, 10];

        phases.iter().zip(weights.iter())
            .take_while(|(p, _)| **p != current_phase)
            .map(|(_, w)| *w)
            .sum()
    }
}
```

### 9.5.2 LRU 缓存策略

```rust
use lru::LruCache;
use std::num::NonZeroUsize;

/// 模型缓存: 最多同时缓存 2 个模型
struct ModelCache {
    cache: LruCache<String, LoadedModel>,
    max_memory_mb: usize,
    current_memory_mb: usize,
}

impl ModelCache {
    fn new(max_memory_mb: usize) -> Self {
        Self {
            cache: LruCache::new(NonZeroUsize::new(2).unwrap()),
            max_memory_mb,
            current_memory_mb: 0,
        }
    }

    fn get_or_load(&mut self, model_id: &str, path: &Path) -> Result<&LoadedModel> {
        if self.cache.contains(model_id) {
            self.cache.get(model_id);  // 触发 LRU 更新
            return Ok(self.cache.get(model_id).unwrap());
        }

        // 检查是否有足够空间
        let model_size = estimate_model_size(path)?;
        while self.current_memory_mb + model_size > self.max_memory_mb {
            // 淘汰最旧的模型
            if let Some((_, evicted)) = self.cache.pop_lru() {
                self.current_memory_mb -= evicted.memory_mb;
                evicted.unload();
            } else {
                return Err(Error::OutOfMemory);
            }
        }

        // 加载新模型
        let model = LoadedModel::load(path)?;
        self.current_memory_mb += model.memory_mb;
        self.cache.put(model_id.to_string(), model);

        Ok(self.cache.get(model_id).unwrap())
    }
}
```

### 9.5.4 OOM 预防

```rust
/// 任务提交前的内存预估检查
fn check_memory_budget(params: &CompareParams, loaded_models: &[ModelInfo]) -> Result<()> {
    let estimated_mb = estimate_compare_memory(params, loaded_models);

    let sys = sysinfo::System::new_all();
    let total_mb = sys.total_memory() / 1024 / 1024;
    let available_mb = (sys.total_memory() - sys.used_memory()) / 1024 / 1024;

    // 分级策略：
    // - 建议级: 预估 > 系统内存 60% → UI 提示但允许继续
    // - 警告级: 预估 > 系统内存 80% → 强烈建议关闭其他应用
    // - 拒绝级: 预估 > 可用内存 - 500MB → 拒绝执行
    let threshold_80 = (total_mb as f64 * 0.80) as u64;
    let threshold_reject = available_mb.saturating_sub(500);

    if estimated_mb > threshold_reject {
        return Err(Error::OutOfMemory {
            required_mb: estimated_mb,
            available_mb,
            message: format!(
                "预估需要 {}MB 内存，当前可用 {}MB（系统总计 {}MB）。请关闭其他应用或缩小文档。",
                estimated_mb, available_mb, total_mb
            ),
        });
    }

    if estimated_mb > threshold_80 {
        // 发送警告但不阻止执行
        warn!(
            "内存预估 {}MB 超过系统内存 80%（{}MB），可能存在 OOM 风险",
            estimated_mb, total_mb
        );
    }

    Ok(())
}

fn estimate_compare_memory(params: &CompareParams, loaded_models: &[ModelInfo]) -> u64 {
    let doc_a_size = params.doc_a.ast.estimate_memory();
    let doc_b_size = params.doc_b.ast.estimate_memory();

    // AST + Chunks + Embeddings + Similarity Matrix + Diff
    let data_mb = ((doc_a_size + doc_b_size) as f64 * 3.5 / 1024.0 / 1024.0) as u64;

    // 加上已加载模型的内存
    let model_mb: u64 = loaded_models.iter().map(|m| m.memory_mb).sum();

    data_mb + model_mb
}
```

### 9.6.1 管道式背压设计

BidLens 的比对流程是一个多阶段管道，每个阶段的输出是下一阶段的输入。使用有界 channel 实现背压控制：

```
│  │  Parser  │──────────────►│  Chunker  │──────────────►│  Embedder │  │
│  │          │               │          │               │           │  │
```

```rust
struct ComparePipeline {
    // Parser → Chunker
    ast_tx: mpsc::Sender<Document>,
    ast_rx: mpsc::Receiver<Document>,

    // Chunker → Embedder
    chunk_tx: mpsc::Sender<Vec<Chunk>>,
    chunk_rx: mpsc::Receiver<Vec<Chunk>>,

    // Embedder → Matcher
    embed_tx: mpsc::Sender<Vec<(Chunk, Vec<f32>)>>,
    embed_rx: mpsc::Receiver<Vec<(Chunk, Vec<f32>)>>,

    // Matcher → Differ
    pair_tx: mpsc::Sender<Vec<MatchPair>>,
    pair_rx: mpsc::Receiver<Vec<MatchPair>>,
}

impl ComparePipeline {
    fn new() -> Self {
        let (ast_tx, ast_rx) = mpsc::channel(16);       // 小缓冲: AST 较大
        let (chunk_tx, chunk_rx) = mpsc::channel(32);    // 中缓冲: Chunk 适中
        let (embed_tx, embed_rx) = mpsc::channel(32);    // 中缓冲: 向量适中
        let (pair_tx, pair_rx) = mpsc::channel(64);      // 大缓冲: 匹配对较小

        Self { ast_tx, ast_rx, chunk_tx, chunk_rx, embed_tx, embed_rx, pair_tx, pair_rx }
    }
```

```rust
/// 管道背压监控指标
struct PipelineMetrics {
    // 各阶段缓冲区使用率
    ast_buffer_usage: AtomicUsize,
    chunk_buffer_usage: AtomicUsize,
    embed_buffer_usage: AtomicUsize,
    pair_buffer_usage: AtomicUsize,

    // 背压触发次数
    backpressure_count: AtomicUsize,

    // 各阶段吞吐率（items/sec）
    parse_throughput: AtomicF64,
    embed_throughput: AtomicF64,
    match_throughput: AtomicF64,
}

impl PipelineMetrics {
    /// 判断是否有背压
    fn has_backpressure(&self) -> bool {
        self.ast_buffer_usage.load(Ordering::Relaxed) > 12  // > 75% of 16
            || self.chunk_buffer_usage.load(Ordering::Relaxed) > 24
            || self.embed_buffer_usage.load(Ordering::Relaxed) > 24
    }

    /// 获取瓶颈阶段
    fn bottleneck(&self) -> &'static str {
        let usages = [
            ("parse", self.ast_buffer_usage.load(Ordering::Relaxed)),
            ("chunk", self.chunk_buffer_usage.load(Ordering::Relaxed)),
            ("embed", self.embed_buffer_usage.load(Ordering::Relaxed)),
            ("match", self.pair_buffer_usage.load(Ordering::Relaxed)),
        ];

        usages.iter()
            .max_by_key(|(_, u)| *u)
            .map(|(name, _)| *name)
            .unwrap_or("none")
    }
}
```

### 9.7.1 IPC 消息路由

```typescript
// Electron 主进程 IPC 路由
class IpcRouter {
  private pendingRequests = new Map<string, {
    resolve: (value: any) => void;
    reject: (error: Error) => void;
    onProgress?: (progress: ProgressEvent) => void;
  }>();

  constructor(private rustBridge: RustBridge) {}

  // 渲染进程调用入口
  async handleCompare(
    docA: DocumentData,
    docB: DocumentData,
    options: CompareOptions,
    onProgress?: (progress: ProgressEvent) => void
  ): Promise<CompareResult> {
    const id = `compare-${uuid()}`;

    return new Promise((resolve, reject) => {
      this.pendingRequests.set(id, { resolve, reject, onProgress });

      // 发送到 Rust 引擎
      this.rustBridge.send({
        id,
        method: 'compare',
        params: { doc_a: docA, doc_b: docB, options },
      });
    });
  }

  // 处理来自 Rust 引擎的消息
  handleMessage(message: JsonRpcMessage) {
    if ('event' in message && message.event === 'progress') {
      // 进度事件
      const pending = this.pendingRequests.get(message.id);
      pending?.onProgress?.(message.data);
      return;
    }

    if ('id' in message) {
      const pending = this.pendingRequests.get(message.id);
      if (!pending) return;

      this.pendingRequests.delete(message.id);

      if ('result' in message) {
        pending.resolve(message.result);
      } else if ('error' in message) {
        pending.reject(new EngineError(message.error));
      }
    }
  }
}
```

```rust
/// 阶段级耗时追踪
struct StageTimer {
    stages: Vec<(String, Duration)>,
    current_stage: Option<(String, Instant)>,
}

impl StageTimer {
    fn start_stage(&mut self, name: &str) {
        self.current_stage = Some((name.to_string(), Instant::now()));
    }

    fn end_stage(&mut self) {
        if let Some((name, start)) = self.current_stage.take() {
            self.stages.push((name, start.elapsed()));
        }
    }

    fn report(&self) -> StageReport {
        let total: Duration = self.stages.iter().map(|(_, d)| *d).sum();

        StageReport {
            stages: self.stages.iter().map(|(name, duration)| {
                StageInfo {
                    name: name.clone(),
                    duration_ms: duration.as_millis() as u64,
                    percent: (duration.as_secs_f64() / total.as_secs_f64() * 100.0) as u8,
                }
            }).collect(),
            total_ms: total.as_millis() as u64,
        }
    }
}

/// 使用示例

    let mut timer = StageTimer::new();

    let (ast_a, ast_b) = parse_documents(&params).await?;
    timer.end_stage();

    timer.start_stage("chunk");
    let chunks_a = chunk_document(&ast_a, &params.options);
    let chunks_b = chunk_document(&ast_b, &params.options);
    timer.end_stage();

    timer.start_stage("embed");
    let embeddings = do_embed(&chunks_a, &chunks_b).await?;
    timer.end_stage();

    timer.start_stage("match");
    let pairs = do_match(&embeddings, params.options.threshold).await?;
    timer.end_stage();

    timer.start_stage("diff");
    let diff_result = do_diff(&pairs, &ast_a, &ast_b).await?;
    timer.end_stage();

    let report = timer.report();
    // 存储到 SQLite 性能指标表
    save_performance_metrics(&report).await?;

    Ok(CompareResult { pairs, stats: report.into() })
}
```

**本章小结**

BidLens 的多线程与任务调度设计遵循以下核心原则：

1. **职责分离**：Tokio 管 IO，Rayon 管 CPU，Electron 管 UI，三者各司其职
2. **背压传播**：有界 channel 确保管道各阶段不会过度积压
3. **协作取消**：CancellationToken 贯穿整个任务生命周期，确保资源正确释放
4. **内存安全**：预估检查 + 压力检测 + LRU 淘汰，三层防护避免 OOM
5. **可观测性**：全链路指标采集，支持性能分析和瓶颈定位
