### 8.2.2 本地 Embedding — ONNX Runtime

**模型加载流程**：

```rust
use ort::{Session, GraphOptimizationLevel, SessionOutputs};

struct EmbeddingEngine {
    session: Option<Session>,
    tokenizer: Tokenizer,
    model_id: String,
    embedding_dim: usize,
    max_seq_length: usize,
}

impl EmbeddingEngine {
    fn load_model(&mut self, model_path: &Path) -> Result<()> {
        let session = Session::builder()?
            .with_optimization_level(GraphOptimizationLevel::Level3)?
            .with_intra_threads(num_cpus::get())?  // 使用所有 CPU 核心
            .with_model_from_file(model_path)?;

        self.session = Some(session);
        self.embedding_dim = 1024; // BGE-M3 输出维度
        self.max_seq_length = 8192; // BGE-M3 最大序列长度

        Ok(())
    }
}
```

**推理管线（Tokenize → Model → Normalize）**：

```rust
async fn embed_texts(&self, texts: &[String]) -> Result<Vec<Vec<f32>>> {
    let session = self.session.as_ref().ok_or(Error::ModelNotLoaded)?;

    // Step 1: Tokenize
    let encodings = self.tokenizer.encode_batch(
        texts.to_vec(),
        true,  // add_special_tokens
    )?;

    // Step 2: 构建输入张量
    let input_ids = encodings.iter()
        .map(|e| e.get_ids().to_vec())
        .collect::<Vec<_>>();
    let attention_mask = encodings.iter()
        .map(|e| e.get_attention_mask().to_vec())
        .collect::<Vec<_>>();

    // Step 3: ONNX 推理
    let input_tensor = Tensor::from_array((input_ids.as_slice(), [texts.len(), max_len]))?;
    let mask_tensor = Tensor::from_array((attention_mask.as_slice(), [texts.len(), max_len]))?;

    let outputs = session.run(ort::inputs!["input_ids" => input_tensor, "attention_mask" => mask_tensor]?)?;

    // Step 4: 提取 [CLS] token 的隐藏状态
    let hidden_states: ndarray::ArrayView3<f32> = outputs[0].try_extract_tensor()?;

    // Step 5: L2 归一化
    let embeddings: Vec<Vec<f32>> = hidden_states.outer_iter()
        .map(|seq| {
            let cls = seq.row(0).to_owned();
            let norm = cls.iter().map(|x| x * x).sum::<f32>().sqrt();
            cls.iter().map(|x| x / norm).collect()
        })
        .collect();

    Ok(embeddings)
}
```

### 8.2.3 外部 Embedding — OpenAI 兼容 API

```rust
struct ExternalEmbeddingClient {
    client: reqwest::Client,
    api_url: String,
    api_key: String,
    model_name: String,
    max_retries: u32,
}

impl ExternalEmbeddingClient {
    async fn embed(&self, texts: &[String]) -> Result<Vec<Vec<f32>>> {
        let request_body = serde_json::json!({
            "model": self.model_name,
            "input": texts,
            "encoding_format": "float"
        });

        let mut last_error = None;

        // 重试策略：指数退避
        for attempt in 0..self.max_retries {
            match self.send_request(&request_body).await {
                Ok(embeddings) => return Ok(embeddings),
                Err(e) => {
                    last_error = Some(e);
                    if attempt < self.max_retries - 1 {
                        let delay = Duration::from_secs(2u64.pow(attempt));
                        tokio::time::sleep(delay).await;
                    }
                }
            }
        }

        // 所有重试失败，降级到本地模型
        Err(last_error.unwrap_or(Error::ApiUnknown))
    }

    async fn send_request(&self, body: &serde_json::Value) -> Result<Vec<Vec<f32>>> {
        let response = self.client
            .post(&self.api_url)
            .header("Authorization", format!("Bearer {}", self.api_key))
            .json(body)
            .timeout(Duration::from_secs(30))
            .send()
            .await?;

        match response.status() {
            StatusCode::OK => {
                let result: EmbeddingResponse = response.json().await?;
                Ok(result.data.into_iter().map(|d| d.embedding).collect())
            }
            StatusCode::TOO_MANY_REQUESTS => {
                // 速率限制，等待后重试
                let retry_after = response.headers()
                    .get("retry-after")
                    .and_then(|v| v.to_str().ok())
                    .and_then(|v| v.parse::<u64>().ok())
                    .unwrap_or(5);
                tokio::time::sleep(Duration::from_secs(retry_after)).await;
                Err(Error::RateLimited)
            }
            status => Err(Error::ApiError(status.as_u16(), response.text().await?))
        }
    }
}
```

**降级策略**：
```

    │ 仍然失败
```

### 8.3.1 主策略：Section-based 切分

```
Document
```

**切分算法**：

```rust
fn chunk_document(ast: &Document, options: &ChunkOptions) -> Vec<Chunk> {
    let mut chunks = Vec::new();
    let mut current_chunk = ChunkBuilder::new();

    for section in ast.sections() {
        let section_tokens = count_tokens(&section.full_text());

        if current_chunk.tokens() + section_tokens <= options.max_chunk_tokens {
            // 当前 section 可以放入当前 chunk
            current_chunk.add_section(section);
        } else if section_tokens > options.max_chunk_tokens {
            // 单个 section 超长，需要拆分
            if current_chunk.tokens() > 0 {
                chunks.push(current_chunk.build());
                current_chunk = ChunkBuilder::new();
            }
            // 递归拆分超长 section
            chunks.extend(split_long_section(section, options));
        } else {
            // 当前 chunk 已满，开始新 chunk
            chunks.push(current_chunk.build());
            current_chunk = ChunkBuilder::new();
            current_chunk.add_section(section);
        }
    }

    if current_chunk.tokens() > 0 {
        chunks.push(current_chunk.build());
    }

    chunks
}
```

```rust
// 表格数据结构化处理
fn chunk_table(table: &Table) -> Vec<Chunk> {
    // 1. 表格标题作为 chunk 开头
    // 2. 每行转为文本: "列名1: 值1, 列名2: 值2, ..."
    // 3. 如果表格超长，按行分块
    // 4. 每个 table chunk 标记 type="table"

    let header_text = table.headers.join(" | ");
    let rows_text: Vec<String> = table.rows.iter()
        .map(|row| {
            table.headers.iter().zip(row)
                .map(|(h, v)| format!("{}: {}", h, v))
                .collect::<Vec<_>>()
                .join(", ")
        })
        .collect();

    // 按 token 限制分块
    split_by_token_limit(&header_text, &rows_text, 512)
}
```

### 8.3.4 Token 计算

```rust
// 快速 Token 估算（不需要完整 tokenizer）
fn estimate_tokens(text: &str) -> usize {
    let chinese_chars = text.chars().filter(|c| c.is_cjk()).count();
    let english_words = text.split_whitespace()
        .filter(|w| w.chars().all(|c| c.is_ascii_alphabetic()))
        .count();
    let other_chars = text.chars().count() - chinese_chars - english_words;

    // 估算规则:
    // - 中文字符: ~1.5 token/字
    // - 英文单词: ~1 token/word
    // - 数字/标点: ~0.5 token/字符
    (chinese_chars as f64 * 1.5 + english_words as f64 + other_chars as f64 * 0.5) as usize
}
```

### 8.3.5 Overlap 实现

```
无 Overlap:
Chunk 1: [AAAAAAAAAA]
Chunk 2:          [BBBBBBBBBB]
Chunk 3:                   [CCCCCCCCCC]

有 Overlap (10%):
Chunk 1: [AAAAAAAAAA]
Chunk 2:       [AAAAAAAAAAABBBBBBBBBB]
Chunk 3:                      [BBBBBBBBBBCCCCCCCCCC]
         ←overlap→            ←overlap→

实现方式:
1. 计算 overlap tokens = max_chunk_tokens * overlap_ratio (512 * 0.1 = 51)
2. 下一个 chunk 的起始位置 = 上一个 chunk 结束位置 - overlap tokens
3. 边界词完整保留: 不在词中间截断
```

```rust
fn apply_overlap(chunks: Vec<Chunk>, overlap_ratio: f64) -> Vec<Chunk> {
    let overlap_tokens = (chunks[0].token_count as f64 * overlap_ratio) as usize;

    let mut overlapped = Vec::new();
    for (i, chunk) in chunks.iter().enumerate() {
        if i == 0 {
            overlapped.push(chunk.clone());
            continue;
        }

        let prev_chunk = &chunks[i - 1];
        // 从前一个 chunk 的末尾取 overlap_tokens 个 token
        let overlap_text = prev_chunk.tail_tokens(overlap_tokens);
        let mut new_chunk = chunk.clone();
        new_chunk.prepend_text(&overlap_text);
        overlapped.push(new_chunk);
    }

    overlapped
}
```

## 8.4 向量相似度计算

### 8.4.1 余弦相似度公式与实现

```
余弦相似度公式:
                    A · B
cos(θ) = ─────────────────────
           ||A|| × ||B||

其中:
- A · B = Σ(a_i × b_i)        (向量点积)
- ||A|| = √Σ(a_i²)           (向量 L2 范数)
- 结果范围: [-1, 1]
- 由于 Embedding 已 L2 归一化，简化为: cos(θ) = A · B
```

```rust
// 归一化后的向量，余弦相似度简化为点积
fn cosine_similarity(a: &[f32], b: &[f32]) -> f32 {
    debug_assert_eq!(a.len(), b.len());
    a.iter().zip(b.iter()).map(|(x, y)| x * y).sum()
}

// 批量计算: A (m×d) 与 B (n×d) 的相似度矩阵
fn batch_cosine_similarity(a: &[Vec<f32>], b: &[Vec<f32>]) -> Vec<Vec<f32>> {
    let m = a.len();
    let n = b.len();
    let d = a[0].len();

    // 使用矩阵乘法加速 (m×d) × (d×n) = (m×n)
    let a_matrix = nalgebra::DMatrix::from_row_slice(m, d, flatten(a));
    let b_matrix = nalgebra::DMatrix::from_row_slice(n, d, flatten(b));

    let sim_matrix = &a_matrix * &b_matrix.transpose();

    // 转换为 Vec<Vec<f32>>
    (0..m).map(|i| {
        (0..n).map(|j| sim_matrix[(i, j)]).collect()
    }).collect()
}
```

### 8.4.2 批量计算优化

```rust
use rayon::prelude::*;

fn parallel_similarity(a: &[Vec<f32>], b: &[Vec<f32>], threshold: f32) -> Vec<(usize, usize, f32)> {
    a.par_iter().enumerate().flat_map(|(i, a_vec)| {
        b.par_iter().enumerate()
            .filter_map(|(j, b_vec)| {
                let sim = cosine_similarity(a_vec, b_vec);
                if sim >= threshold {
                    Some((i, j, sim))
                } else {
                    None
                }
            })
            .collect::<Vec<_>>()
    }).collect()
}
```

### 8.4.3 Top-K 选择算法

```rust
// 使用最小堆获取 Top-K 匹配
fn top_k_matches(similarities: &[(usize, usize, f32)], k: usize) -> Vec<(usize, usize, f32)> {
    let mut heap = BinaryHeap::with_capacity(k);

    for &(i, j, sim) in similarities {
        if heap.len() < k {
            heap.push(Reverse(OrdF32(sim), i, j));
        } else if sim > heap.peek().unwrap().0 .0 {
            heap.pop();
            heap.push(Reverse(OrdF32(sim), i, j));
        }
    }

    heap.into_iter().map(|Reverse(sim, i, j)| (i, j, sim.0)).collect()
}
```

### 8.5.3 动态阈值建议

```rust
fn suggest_threshold(similarities: &[f32]) -> f32 {
    // 统计相似度分布
    let mean = similarities.iter().sum::<f32>() / similarities.len() as f32;
    let std_dev = {
        let variance = similarities.iter()
            .map(|s| (s - mean).powi(2))
            .sum::<f32>() / similarities.len() as f32;
        variance.sqrt()
    };

    // 建议阈值 = 均值 - 1 标准差（保留大部分合理匹配）
    let suggested = (mean - std_dev).max(0.5).min(0.9);

    // 四舍五入到 0.05 的倍数
    (suggested * 20.0).round() / 20.0
}
```

### 8.6.1 一对多匹配（A 段拆分匹配）

当文档 A 的一段内容在文档 B 中被拆分为多段时：

```
1. 对 A 的每个 chunk，找到所有相似度 > threshold 的 B chunk

3. 合并显示: A 的原文 vs B 多段的拼接原文

```

```
检测算法:
1. 对 B 的每个 chunk，找到所有相似度 > threshold 的 A chunk
2. 如果一个 B chunk 匹配到多个 A chunk，标记为 many_to_one
3. 合并显示: A 多段的拼接原文 vs B 的原文
```

### 8.6.3 顺序调整检测

```rust
fn detect_order_changes(pairs: &[(usize, usize, f32)]) -> Vec<OrderChange> {
    let mut changes = Vec::new();

    for (i, &(a_idx, b_idx, _)) in pairs.iter().enumerate() {
        // 计算位置差异
        let position_diff = (a_idx as i32) - (b_idx as i32);

        if position_diff.abs() > 2 {  // 位置偏移超过 2 个 chunk
            changes.push(OrderChange {
                a_index: a_idx,
                b_index: b_idx,
                direction: if position_diff > 0 { "moved_down" } else { "moved_up" },
                distance: position_diff.abs() as usize,
            });
        }
    }

    changes
}
```

### 8.6.4 碎片匹配聚类

```
当多个低相似度的匹配聚集在一起时，可能是同一内容的碎片化表达:

碎片匹配检测:
1. 收集所有相似度在 [threshold-0.1, threshold) 区间的候选对
2. 按 A chunk 索引排序
3. 如果连续 3+ 个 A chunk 都有候选匹配，标记为碎片匹配
4. 聚类后合并显示，标注 "碎片匹配 (置信度: 低)"

示例:
A0 ↔ B5 (sim=0.70)  ─┐
A1 ↔ B6 (sim=0.72)   ├── 碎片匹配组
A2 ↔ B7 (sim=0.68)  ─┘
```

### 8.6.5 匹配对冲突解决

当多个匹配方案冲突时（一个 chunk 匹配到多个候选），使用 **贪心匹配 + 置信度优先** 策略：

```rust
fn resolve_conflicts(candidates: Vec<(usize, usize, f32)>, threshold: f32) -> Vec<(usize, usize, f32)> {
    // 1. 按相似度降序排序
    let mut sorted = candidates;
    sorted.sort_by(|a, b| b.2.partial_cmp(&a.2).unwrap());

    // 2. 贪心选择: 优先选择相似度最高的匹配对
    let mut used_a = HashSet::new();
    let mut used_b = HashSet::new();
    let mut result = Vec::new();

    for (a_idx, b_idx, sim) in sorted {
        if sim < threshold { break; }
        if !used_a.contains(&a_idx) && !used_b.contains(&b_idx) {
            used_a.insert(a_idx);
            used_b.insert(b_idx);
            result.push((a_idx, b_idx, sim));
        }
    }

    result
}
```

### 8.9.1 模型下载

```typescript
interface DownloadConfig {
  url: string;              // 下载地址
  dest_path: string;        // 目标路径
  expected_sha256: string;  // 预期 SHA256
  expected_size: number;    // 预期文件大小
  chunk_size: number;       // 分块大小（默认 1MB）
  max_retries: number;      // 最大重试次数（默认 3）
}

// 下载流程
async function downloadModel(config: DownloadConfig): Promise<void> {
  // 1. 检查是否已下载（SHA256 匹配）
  if (await verifyExistingFile(config.dest_path, config.expected_sha256)) {
    return; // 已下载，跳过
  }

  // 2. 检查磁盘空间
  const available = await getAvailableDiskSpace();
  if (available < config.expected_size * 1.2) {
    throw new Error('磁盘空间不足');
  }

  // 3. 断点续传下载
  const tempPath = config.dest_path + '.downloading';
  const downloaded = await getFileSize(tempPath).catch(() => 0);

  const response = await fetch(config.url, {
    headers: { 'Range': `bytes=${downloaded}-` }
  });

  // 4. 流式写入 + 进度报告
  const writer = fs.createWriteStream(tempPath, { flags: 'a' });
  let totalDownloaded = downloaded;

  for await (const chunk of response.body) {
    writer.write(chunk);
    totalDownloaded += chunk.length;

    // 报告进度
    reportProgress({
      model_id: config.model_id,
      percent: (totalDownloaded / config.expected_size) * 100,
      speed_mbps: calculateSpeed(),
    });
  }

  // 5. SHA256 校验
  const sha256 = await calculateSHA256(tempPath);
  if (sha256 !== config.expected_sha256) {
    await fs.unlink(tempPath);
    throw new Error('SHA256 校验失败');
  }

  // 6. 重命名（原子操作）
  await fs.rename(tempPath, config.dest_path);
}
```
